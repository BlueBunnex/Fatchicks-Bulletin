SETUP LOCAL REPO
    git init
    git remote add origin <URL>
    
    PULL
    git fetch origin main
    git reset --hard FETCH_HEAD
    
    PUSH
    git add <. / -all / PATH> [help]
    git commit -m "<MESSAGE>"
    git push -u origin main
    
    JAR
    unzip <JAR>.jar -d <DIR>
    jar cvf <JAR>.jar -C <DIR>/ .
